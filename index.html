cache_or_call <- function(filename, fun) {
    if (file.exists(filename)) {
        readRDS(filename)
    } else {
        tmp <- fun()
        saveRDS(tmp, filename)
        tmp
    }
}

Pulling the health facility match up result and define the “Best” LGAs

# Loading matching status data from nmis dump
require(RJSONIO)
baseline_status <- fromJSON("~/Dropbox/Nigeria/Nigeria 661 Baseline Data Cleaning/in_process_data/mop_up_matching_result/Matching status/nmis_by_lga.json", 
    simplifyWithNames = T)

baseline_status <- do.call("rbind.fill", lapply(baseline_status, as.data.frame, 
    stringsAsFactors = F))

baseline_status <- subset(baseline_status, select = c("lga_id", "health_total", 
    "health_finished", "health_matched"))
baseline_status <- rename(baseline_status, c(health_total = "baseline_total", 
    health_matched = "matched", health_finished = "baseline_finished"))


facility_status <- fromJSON("~/Dropbox/Nigeria/Nigeria 661 Baseline Data Cleaning/in_process_data/mop_up_matching_result/Matching status/lga_by_lga.json", 
    simplifyWithNames = T)

facility_status <- do.call("rbind.fill", lapply(facility_status, as.data.frame, 
    stringsAsFactors = F))

facility_status <- subset(facility_status, select = c("lga_id", "health_total", 
    "health_finished"))
facility_status <- rename(facility_status, c(health_total = "facility_total", 
    health_finished = "facility_finished"))


# Load lga info
lgas <- read.csv("~/Dropbox/Nigeria/Nigeria 661 Baseline Data Cleaning/lgas.csv")

edu <- merge(lgas, baseline_status, by = "lga_id", all = T)
edu <- merge(edu, facility_status, by = "lga_id", all = T)


edu_state <- ddply(edu, .(state), summarize, match_count = sum(matched, na.rm = T), 
    baseline_total = sum(baseline_total, na.rm = T), baseline_finished = sum(baseline_finished, 
        na.rm = T), facility_total = sum(facility_total, na.rm = T), facility_finished = sum(facility_finished, 
        na.rm = T))

edu_state$base_facil <- edu_state$baseline_total/edu_state$facility_total
edu_state$match_facil <- edu_state$match_count/edu_state$facility_total
edu_state$facil_complete <- edu_state$facility_finished/edu_state$facility_total
edu_state$base_complete <- edu_state$baseline_finished/edu_state$baseline_total

edu_state$idx <- edu_state$base_facil + edu_state$match_facil + edu_state$facil_complete

edu_state$idx2 <- (0.7 * order(edu_state$base_facil, decreasing = F) + 0.5 * 
    order(edu_state$match_facil, decreasing = F) - 0.2 * order(edu_state$match_facil, 
    decreasing = T))

edu_state <- arrange(edu_state, desc(idx2))
