cache_or_call <- function(filename, fun) {
    if (file.exists(filename)) {
        readRDS(filename)
    } else {
        tmp <- fun()
        saveRDS(tmp, filename)
        tmp
    }
}

ahd <- subset(as.data.frame(akwa_health), select = c("x", "y", "short_id"))
ahd$short_id <- as.character(ahd$short_id)

nearestH <- pick.from.points(popPoints, ahd, radius = 2e+10)
merged <- merge(nearestH, ahd, by = "short_id", suffixes = c(".pop", ".health"))
merged$N <- row.names(merged)
mergedPlotDF <- ddply(idata.frame(merged), .(N), function(df) {
    data.frame(x = c(df$x.pop, df$x.health), y = c(df$y.pop, df$y.health), group = c(df[1, 
        "N"], df[1, "N"]), type = c("Population Point", "Health Facility"))
})
gplot(nga_akwa_raster) + geom_tile(aes(fill = value)) + scale_fill_continuous(low = "white", 
    trans = "log10") + geom_line(data = mergedPlotDF, aes(x = x, y = y, group = group))
